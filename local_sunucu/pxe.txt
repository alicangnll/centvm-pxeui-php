    Installer

    MkFile
    mkdir /var/lib/tftpboot/data
    mkdir /var/lib/tftpboot/data/CentOS7
    mkdir /var/lib/tftpboot/data/CentOS7/backup
    mkdir /mnt/CentOS7
    mkdir /var/lib/tftpboot/data/freedos
    mkdir /var/lib/tftpboot/data/Windows
    

    DNSMASQ Settings
    cp --backup /etc/dnsmasq.conf /var/lib/tftpboot/data/CentOS7/backup/dnsmasq.conf-bk
    echo  # DNS Options
port=0
interface=enp0s17,lo


# DHCP Options
dhcp-range=10.0.2.1,proxy
dhcp-option=vendor:PXEClient,6,2b
dhcp-no-override


# PXE
dhcp-boot=pxelinux.0


#pxe-prompt="Press F8 for menu.", 10
pxe-service=X86PC, "alicangonullu.xyz", pxelinux


# TFTP Options
enable-tftp
tftp-root=/var/lib/tftpboot >> /etc/dnsmasq.conf
    

    Syslinux Settings
    mkdir /var/lib/tftpboot/pxelinux.cfg
    cp -v /usr/share/syslinux/* /var/lib/tftpboot
    

    Firewall Settings
    firewall-cmd --add-service=http --zone=public --permanent
    firewall-cmd --add-service=tftp --zone=public --permanent
    firewall-cmd --add-service=samba --zone=public --permanent
    firewall-cmd --reload
    

    TFTP Settings
    mkdir /var/lib/tftpboot/data/CentOS7
    chmod -R 755 /var/lib/tftpboot/data
    

    HTTPD Settings
    touch /etc/httpd/conf.d/pxeboot.conf
    chmod 770 /var/lib/tftpboot/data
    chown apache. /var/lib/tftpboot/data
    chcon -R -t httpd_sys_rw_content_t /var/lib/tftpboot/data
    semanage fcontext -a -t httpd_sys_rw_content_t /var/lib/tftpboot/data
    echo 
<IfModule mod_dav_fs.c>
    DAVLockDB /var/lib/dav/lockdb
</IfModule>
<VirtualHost *:80>
ServerAdmin webmaster@localhost
DocumentRoot /var/www/html
Alias /pxeboot /var/lib/tftpboot/data
<Directory /var/lib/tftpboot/data>
DAV On
Options Indexes FollowSymLinks
Require all granted
</Directory>
</VirtualHost> >> /etc/httpd/conf.d/pxeboot.conf
    

    Firewall Process Two
    systemctl stop firewalld
    systemctl disable firewalld
    

    Syslinux Process Two
    setenforce 0 && sed -i 's/SELINUX=enforcing/SELINUX=permissive/g' /etc/sysconfig/selinux
    

    NFS Server Install
    echo "/var/lib/tftpboot/data  *(rw)" >> /etc/exports
    

    TFTP Server Install
    touch /etc/xinetd.d/tftp
    cp --backup /etc/xinetd.d/tftp /var/lib/tftpboot/data/CentOS7/backup/tftp-bk
    echo #
service tftp
{
	socket_type		= dgram
	protocol		= udp
	wait			= yes
	user			= root
	server			= /usr/sbin/in.tftpd
	server_args		= -c /var/lib/tftpboot
	disable			= no
	per_source		= 11
	cps			= 100 2
	flags			= IPv4
} >> /etc/xinetd.d/tftp
    

    TFTP Server Start
    systemctl restart tftp
    systemctl enable tftp
    

    NFS Server Start
    systemctl restart nfs-server
    systemctl enable nfs-server
    

    XÄ°net Server Start
    systemctl restart xinetd
    systemctl enable xinetd
